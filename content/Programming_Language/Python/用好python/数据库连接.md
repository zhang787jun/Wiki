确保数据库连接关闭（其它类似资源也可以这样实现）
```python

def with_db(db: Connection, exception_callback=None):
    def decorator(f):
        def db_context(*a, **kw):
            try:
                return f(db, *a, **kw)
            except Exception as e:
                logging.exception(str(e))
                if exception_callback is not None:
                    exception_callback(e)
            finally:
     
           try:
                    db.close()
                except:
                    pass
        return db_context

    return decorator
```
