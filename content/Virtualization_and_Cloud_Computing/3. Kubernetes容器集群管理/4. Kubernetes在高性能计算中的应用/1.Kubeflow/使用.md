```shell
cat << EOF > notebook_pv.yaml
apiVersion: v1
kind: PersistentVolume
metadata:
  name: notebook-pv
  namespace: mykubeflow1
  labels:
    type: local
    app: notebook-pv
    key: notebook-pv
spec:
  capacity:
    storage: 35Gi
  accessModes:
    - ReadWriteOnce
  hostPath:
    path: /data/notebook-pv
    type: DirectoryOrCreate
EOF
kubectl create -f notebook_pv.yaml
```
